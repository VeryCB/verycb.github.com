A simple start

var http = require('http');

http.createServer(function(request, response) {
    var requestHeaders = {
        'Content-Type': 'text/html'
    };

    response.writeHead(200, requestHeaders);
    response.write('Hello, this is an example');
    response.end();
}).listen(8080);

console.log('Listening on port 8080...');
