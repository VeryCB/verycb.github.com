批量转换string field为float

var cursor = db.collection.find();
while (cursor.hasNext()) {
    var doc = cursor.next();
    db.collection.update(
        {_id:doc._id}, 
        {$set: {
            field: parseFloat(doc.field)
        }
    });
}
